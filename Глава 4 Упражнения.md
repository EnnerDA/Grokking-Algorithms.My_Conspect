from copy import copy

def sum(user_list):
    mod_user_list = copy(user_list)
    if len(mod_user_list) == 1:
        print(mod_user_list[0])
    else:
        mod_user_list[0] += mod_user_list.pop(1)
        sum(mod_user_list) # Внимание !!! РЕКУРСИЯ па-бааам!
